27 MBMS_RT::Service::Service(const libconfig::Config& cfg, std::string tmgi, const std::string& mcast, unsigned long long tsi, std::string iface, boost::asio::io_service& io_service)
28  : _cfg(cfg)
29  , _tmgi(std::move(tmgi))
30  , _tsi(tsi)
31  , _iface(std::move(iface))
32  , _flute_thread{}
33 {
34  size_t delim = mcast.find(':');
35  if (delim == std::string::npos) {
36  spdlog::error("Invalid multicast address {}", mcast);
37  return;
38  }
39  _mcast_addr = mcast.substr(0, delim);
40  _mcast_port = mcast.substr(delim + 1);
41  spdlog::info("Starting FLUTE receiver on {}:{} for TSI {}", _mcast_addr, _mcast_port, _tsi);
42  _flute_thread = std::thread{[&](){
43  _flute_receiver = std::make_unique<LibFlute::Receiver>(_iface, _mcast_addr, atoi(_mcast_port.c_str()), _tsi, io_service) ;
44  }};
45 }

64 auto MBMS_RT::Service::tryParseBootstrapFile(std::string str) -> void
65 {
66  g_mime_init();
67 
68  // R&S header has no newlines, fix that
69  str = std::regex_replace(str, std::regex(" Content-Type:"), "\nContent-Type:");
70  str = std::regex_replace(str, std::regex(" Content-Description:"), "\nContent-Description:");
71 
72  auto stream = g_mime_stream_mem_new_with_buffer(str.c_str(), str.length());
73  auto parser = g_mime_parser_new_with_stream(stream);
74  g_object_unref(stream);
75 
76  auto mpart = g_mime_parser_construct_part(parser, nullptr);
77  g_object_unref(parser);
78 
79  auto iter = g_mime_part_iter_new(mpart);
80  auto ct = g_mime_object_get_content_type_parameter(mpart, "type");
81  while (g_mime_part_iter_next(iter)) {
82  GMimeObject *current = g_mime_part_iter_get_current (iter);
83  GMimeObject *parent = g_mime_part_iter_get_parent (iter);
84 
85  if (GMIME_IS_MULTIPART (parent) && GMIME_IS_PART (current)) {
86  auto type = std::string(g_mime_content_type_get_mime_type(g_mime_object_get_content_type(current)));
87  auto options = g_mime_format_options_new();
88  g_mime_format_options_add_hidden_header(options, "Content-Type");
89  g_mime_format_options_add_hidden_header(options, "Content-Transfer-Encoding");
90  g_mime_format_options_add_hidden_header(options, "Content-Location");
91 
92  if (type == "application/mbms-user-service-description+xml") {
93  _service_description = g_mime_object_to_string(current, options);
94  } else if (type == "application/sdp" && _sdp.empty()) {
95  _sdp = g_mime_object_to_string(current, options);
96  _sdp = std::regex_replace(_sdp, std::regex("^\n"), "");
97  } else if (type == "application/vnd.apple.mpegurl") {
98  _m3u = g_mime_object_to_string(current, options);
99  }
100  }
101  }
102 
103  if (!_service_description.empty()) {
104  tinyxml2::XMLDocument doc;
105  doc.Parse(_service_description.c_str());
106  auto bundle = doc.FirstChildElement("bundleDescription");
107  if (bundle) {
108  auto usd = bundle->FirstChildElement("userServiceDescription");
109  if (usd) {
110  _service_name = usd->FirstChildElement("name")->GetText();
111  spdlog::debug("Service Name: {}", _service_name);
112  }
113  }
114  }
115 
116  if (!_sdp.empty()) {
117  std::istringstream iss(_sdp);
118  for (std::string line; std::getline(iss, line); )
119  {
120  const std::regex sdp_line_regex("^([a-z])\\=(.+)$");
121  std::smatch match;
122  if (std::regex_match(line, match, sdp_line_regex)) {
123  if (match.size() == 3) {
124  auto field = match[1].str();
125  auto value = match[2].str();
126  spdlog::debug("{}: {}", field, value);
127 
128  if (field == "c") {
129  const std::regex value_regex("^IN (IP.) ([0-9\\.]+).*$");
130  std::smatch cmatch;
131  if (std::regex_match(value, cmatch, value_regex)) {
132  if (cmatch.size() == 3) {
133  _stream_mcast_addr = cmatch[2].str();
134  }
135  }
136  } else if (field == "m") {
137  const std::regex value_regex("^application (.+) (.+)$");
138  std::smatch cmatch;
139  if (std::regex_match(value, cmatch, value_regex)) {
140  if (cmatch.size() == 3) {
141  _stream_mcast_port = cmatch[1].str();
142  _stream_type = cmatch[2].str();
143  }
144  }
145  const std::regex value_regex2("^application (.+) (.+) (.+)$");
146  if (std::regex_match(value, cmatch, value_regex2)) {
147  if (cmatch.size() == 4) {
148  _stream_mcast_port = cmatch[1].str();
149  _stream_type = cmatch[2].str();
150  }
151  }
152  } else if (field == "a") {
153  const std::regex value_regex("^flute-tsi:(.+)$");
154  std::smatch cmatch;
155  if (std::regex_match(value, cmatch, value_regex)) {
156  if (cmatch.size() == 2) {
157  _stream_flute_tsi = stoul(cmatch[1].str());
158  }
159  }
160  }
161  }
162  }
163  }
164 
165  if (_stream_type != "none") {
166  _bootstrap_file_parsed = true;
167  }
168 
169  }
