34 void MBMS_RT::Middleware::tick_handler()
35 {
36  auto mchs = _rp.getMchInfo();
37  _available_services.clear();
38  for (auto const &mch : mchs.as_array()) {
39  for (auto const &mtch : mch.at("mtchs").as_array()) {
40  auto tmgi = mtch.at("tmgi").as_string();
41  auto dest = mtch.at("dest").as_string();
42  _available_services[tmgi] = dest;
43  auto is_service_announcement = std::stoul(tmgi.substr(0,6), nullptr, 16) < 0xF;
44  if (!dest.empty() && is_service_announcement && _services.find(tmgi) == _services.end()) {
45  // automatically start receiving the service announcement
46  // 26.346 5.2.3.1.1 : the pre-defined TSI value shall be "0".
47  _services[tmgi] = std::make_unique<MBMS_RT::Service>(_cfg, tmgi, dest, 0 /*TSI*/, _interface, _io_service);
48  _services[tmgi]->setIsServiceAnnouncement(true);
49  }
50  }
51  }
52 
53  for (auto it = _services.cbegin(); it != _services.cend();)
54  {
55  if (_available_services.find(it->first) == _available_services.end()) {
56  it = _services.erase(it);
57  } else {
58  ++it;
59  }
60  }
61 
62  for (auto const& [tmgi, service] : _services) {
63  // read file lists
64  auto files = service->fileList();
65  for (const auto& file : files) {
66  //_downloaded_files.insert_or_assign(file.location(), file);
67  if (file->meta().content_location == "bootstrap.multipart" && service->isServiceAnnouncement()) {
68  if (!service->bootstrapped()) {
69  service->tryParseBootstrapFile(file->buffer());
70  }
71  }
72  }
73 
74  if (service->bootstrapped() && service->streamType() == "FLUTE/UDP") {
75  if (_payload_flute_streams.count(service->streamMcast()) == 0) {
76  auto mcast_target = service->streamMcast();
77  auto service_available = std::find_if(_available_services.begin(), _available_services.end(),
78  [&mcast_target](const std::pair<std::string, std::string> &p) {
79  return p.second == mcast_target;
80  });
81 
82  if (service_available != _available_services.end()) {
83  if (_services.find(service_available->first) == _services.end()) {
84  service->setStreamTmgi(service_available->first);
85  _services[service_available->first] = std::make_unique<MBMS_RT::Service>(_cfg, service_available->first, service->streamMcast(), service->streamFluteTsi(), _interface, _io_service);
86  }
87  }
88  }
89  }
90  service->remove_expired_files(_max_cache_file_age);
91  }
92 
93  _timer.expires_at(_timer.expires_at() + _tick_interval);
94  _timer.async_wait(boost::bind(&Middleware::tick_handler, this)); //NOLINT
95 }
